<script>
  const poems = {
    "نزار قباني": [
      "علّمني حبكِ أن أحزن، وأنا محتاج منذُ عصورْ...",
      "أحبك جداً وأعرف أن الطريق إلى المستحيل طويل"
    ],
    "غزل": [
      "أنا لي في حبكِ فلسفة، تجعلني ملكاً حين أراكِ",
      "يا زهرةً بين الشفاه، يا همسها في المسامع"
    ],
    "حكمة": [
      "لا تحزن إن خانك من تحب، بل اعرف كيف تختار قلبك من جديد",
      "العقل زينة، ومن خسره عاش في ظلام"
    ]
  };

  function showPoems(key) {
    document.getElementById('layer1').classList.remove('active');
    document.getElementById('layer2').classList.add('active');

    const container = document.getElementById('poemsContainer');
    container.innerHTML = '<h3>' + key + '</h3>';

    poems[key].forEach(text => {
      const div = document.createElement('div');
      div.className = 'poem';
      div.textContent = text;
      container.appendChild(div);
    });
  }

  function goBack() {
    document.getElementById('layer2').classList.remove('active');
    document.getElementById('layer1').classList.add('active');
  }
</script>
