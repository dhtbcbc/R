      ]
    };

    function showPoems(key) {
      document.getElementById('layer1').classList.remove('active');
      document.getElementById('layer2').style.display = 'block';

      const container = document.getElementById('poemsContainer');
      container.innerHTML = '<h3>' + key + '</h3>';

      poems[key].forEach(text => {
        const div = document.createElement('div');
        div.className = 'poem';
        div.textContent = text;
        container.appendChild(div);
      });
    }

    function goBack() {
      document.getElementById('layer2').style.display = 'none';
      document.getElementById('layer1').classList.add('active');
    }
